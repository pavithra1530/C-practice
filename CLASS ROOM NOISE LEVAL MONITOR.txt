#include <stdio.h>
int main() {
    int N, noise;
    int maxNoise = 0;
    int noisyPeriods = 0;
    scanf("%d", &N);
    for(int i = 0; i < N; i++) {
        scanf("%d", &noise);
        if(i == 0) {
            maxNoise = noise;
        }
        if(noise > maxNoise) {
            maxNoise = noise;
        }

        if(noise > 70) {
            noisyPeriods++;
        }
    }
    printf("Maximum Noise: %d\n", maxNoise);
    printf("Noisy Periods: %d", noisyPeriods);

    return 0;
}
