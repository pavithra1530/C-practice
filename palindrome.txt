#include <stdio.h>

int main() {
    int N, original, reverse = 0, digit;

    printf("Enter a number: ");
    scanf("%d", &N);
original = N;
 for (; N > 0; N = N / 10) {
        digit = N % 10;
        reverse = reverse * 10 + digit;
    }if (original == reverse)
        printf("Yes");
    else
        printf("No");

    return 0;}
