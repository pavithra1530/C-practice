#include <stdio.h>
int main() {
    int capacity, N, items;
    int overflowDays = 0;
    scanf("%d", &capacity);
    scanf("%d", &N);

    int remainingCapacity = capacity;

    for(int i = 0; i < N; i++) {
        scanf("%d", &items);

        remainingCapacity -= items;

        if(remainingCapacity < 0) {
            overflowDays++;
        }
    }
    printf("Remaining Capacity: %d\n", remainingCapacity);
    printf("Overflow Days: %d", overflowDays);
    return 0;
}
