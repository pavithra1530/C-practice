#include <stdio.h>

int main() {
    int binary, digit, decimal = 0, base = 1;

    printf("Enter a binary number: ");
    scanf("%d", &binary);

    for (; binary > 0; binary = binary / 10) {
        digit = binary % 10;
        decimal = decimal + digit * base;
        base = base * 2;
    }

    printf("Decimal value = %d", decimal);

    return 0;
}
