#include <stdio.h>
int main() {
    int N;
    long revenue;
    long totalRevenue = 0;
    int targetDays = 0;
    scanf("%d", &N);
    for(int i = 0; i < N; i++) {
        scanf("%ld", &revenue);
        totalRevenue += revenue;

        if(revenue > 50000) {
            targetDays++;
        }
    }
    printf("Total Revenue: %ld\n", totalRevenue);
    printf("Target Days: %d", targetDays);

    return 0;
}
