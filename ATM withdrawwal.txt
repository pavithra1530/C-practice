#include <stdio.h>
int main() {
    int N, amount;
    int total = 0;
    printf("Enter number of withdrawals: ");
    scanf("%d", &N);
    printf("Enter withdrawal amounts:\n");
    for (int i = 1; i <= N; i++) {
        scanf("%d", &amount);
        total += amount;
    }
if (total > 10000)
        printf("Limit Exceeded");
    else
        printf("Approved");

    return 0;
}
